#include<stdio.h>
#define maxn 10009
int f[maxn];
int main()
{
	int C, i, n;
	f[1]=2;
	scanf("%d", &C);
	for(i=2; i<maxn; i++)
	{
		f[i]=f[i-1]+4*(i-1)+1;
	}
	while(C--)
	{
		scanf("%d", &n);
		printf("%d\n", f[n]);
	}	
	return 0;
}
