贪心
题意：n个瓶子，每个瓶子装有一些水，问是否能讲所有的水倒进两个瓶子里面
起初以为是倒水问题，后来才发现是简单的贪心
比较总水量和容量最大的两个瓶子总容量。
O(n)，注意数据范围

#include<stdio.h>
int main()
{
	int n, i;
	long long sum, m1, m2, cur;
	sum=0;
	scanf("%d", &n);
	for(i=0; i<n; i++)
	{
		scanf("%I64d", &cur);
		sum+=cur;
	}
	m1=0, m2=0;
	for(i=0; i<n; i++)
	{
		scanf("%I64d", &cur);
		if(cur>=m1)
		{
			m2=m1;
			m1=cur;
		}
		else
		{
			if(cur>m2)
			{
				m2=cur;
			}
		}
	}
	if(m1+m2>=sum)	printf("YES\n");
	else printf("NO\n");
	return 0;
}
